# Comandos de Diretórios e Arquivos  

```sh

# Verificar Diretório.
pwd

# Listar Diretórios.
ls
ls -lha

# Listar Árvore de Diretórios.
tree

# Mudar de diretório.
cd
cd /home
cd ..
cd ../../teste
cd ~
cd -

# Criar Diretórios.
mkdir teste
mkdir teste teste2
mkdir -p /home/teste

# Remover Diretórios.
rm -rf diretorio
rmdir diretorio_vazio
```
